# Minecraft用
```xml
<repositories>
    <repository>
        <id>github</id>
        <name>NKLab repository</name>
        <url>https://maven.pkg.github.com/Monster2408/repo.mlserver.xyz</url>
    </repository>
</repositories>

<dependencies>
<!-- 旧バージョン(廃止) -->
<dependency>
    <groupId>xyz.mlserver</groupId>
    <artifactId>mc-util</artifactId>
    <version>1.0</version>
</dependency>
<!-- 新バージョン -->
<dependency>
    <groupId>xyz.mlserver</groupId>
    <artifactId>minecraft</artifactId>
    <version>1.1</version>
</dependency>
</dependencies>
```
